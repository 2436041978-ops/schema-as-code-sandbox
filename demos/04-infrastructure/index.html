package intent.governance

default allow = false

# 规则由 Schema Compiler 从 human-ai-boundary.yaml 编译生成
allow {
  input.intent_id == "destructive-action"
  input.human_confirmed == true
  input.actor_type == "human"
}

deny_reason = "AI prohibited from executing destructive actions" {
  input.intent_id == "destructive-action"
  input.actor_type == "ai"
}

# 版本校验：拒绝未知版本的意图协议
allow {
  input.schema_version == "v1.1.0"
}
正交穿透验证：Envoy Sidecar 或 OPA Gatekeeper 加载 Rego 策略文件。容器间通信时自动校验意图协议版本和 Actor 身份，业务 Pod 无感知。
